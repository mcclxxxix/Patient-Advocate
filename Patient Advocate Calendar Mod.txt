class EventPriority(str, Enum):
CRITICAL = "critical" HIGH = "high" MEDIUM = "medium" LOW = "low" # Chemo infusion, surgery — never auto-move
# Oncologist appointment — require patient approval to move
# Lab draw, imaging — can be rescheduled with approval
# Rest day, exercise — AI can suggest moves
class ConflictType(str, Enum):
FATIGUE_WINDOW = "fatigue_window" NEUTROPENIA_RISK = "neutropenia_risk" # Post-chemo nadir period
# Low ANC period — avoid crowds/exposure
DRUG_INTERACTION = "drug_interaction" # Scheduling conflict with med timing
TRAVEL_BURDEN = "travel_burden" # Too many appointments in one day
PATIENT_PREFERENCE = "patient_preference" # Patient-stated unavailability
@dataclass
class CalendarEvent:
event_id: str
title: str
datetime_start: datetime
datetime_end: datetime
priority: EventPriority
location: Optional[str] = None
provider: Optional[str] = None
notes: Optional[str] = None
requires_transport: bool = False
patient_approved: bool = False

def to_dict(self) -> Dict[str, Any]:
return {
"event_id": self.event_id,
"title": self.title,
"start": self.datetime_start.isoformat(),
"end": self.datetime_end.isoformat(),
"priority": self.priority.value,
"location": self.location,
"provider": self.provider,
"notes": self.notes,
"requires_transport": self.requires_transport,
"patient_approved": self.patient_approved,
"created_by": self.created_by,
}

# 3. CALENDAR INTELLIGENCE ENGINE
# ─────────────────────────────────────────────
class CalendarIntelligenceEngine:
"""
Core reasoning engine for schedule optimization.
Clinical logic grounded in:
- Chemotherapy nadir windows (typically days 7-14 post-infusion for ANC)
- Fatigue scoring from symptom logs vs. activity demand
- MOTOR model concept: time-to-event risk stratification (Steinberg et al., 2023)
"""
# Post-chemo nadir window in days (configurable per regimen)
NADIR_WINDOW_DAYS = {"default": (7, 14), "CHOP": (7, 12), "FOLFOX": (9, 14)}
def detect_conflicts(
self,
events: List[CalendarEvent],
lab_values: Optional[Dict],
symptom_log: List[Dict],
patient_profile: Dict,
) -> List[SchedulingSuggestion]:
"""
Detect scheduling conflicts using clinical + logistical heuristics.
Returns a list of suggestions (never direct mutations).
"""
suggestions = []
regimen = patient_profile.get("regimen", "default")
nadir_start, nadir_end = self.NADIR_WINDOW_DAYS.get(
regimen, self.NADIR_WINDOW_DAYS["default"]
)
# Find chemo infusion events as reference anchors
chemo_events = [
e for e in events
if "chemo" in e.title.lower() or "infusion" in e.title.lower()
]
for event in events:
if event.priority == EventPriority.CRITICAL:
continue # Never touch critical events
for chemo in chemo_events:
days_after = (event.datetime_start - chemo.datetime_start).days
# Check nadir window overlap
if nadir_start <= days_after <= nadir_end:
if event.priority in [EventPriority.MEDIUM, EventPriority.LOW]:
# Suggest moving to after nadir recovery
proposed = chemo.datetime_start + timedelta(days=nadir_end + 3)
suggestions.append(SchedulingSuggestion(
suggestion_id=f"sug_{event.event_id}_nadir",
original_event=event,
proposed_new_datetime=proposed,
reason=(
f"'{event.title}' falls within your post-{regimen} nadir wind
f"(days {nadir_start}-{nadir_end}), when immune function is l
f"Exposure risk is elevated."
),
conflict_type=ConflictType.NEUTROPENIA_RISK,
confidence=0.82,
scholarly_basis=(
"CADTH (2024) AI-based patient flow management; "
"Hailey (2024, Oncol. Practice Mgmt.) on adherence-preserving
),
))
# Fatigue-based scheduling (from symptom log)
if symptom_log:
recent_fatigue = [
s["fatigue_score"] for s in symptom_log[-7:]
if "fatigue_score" in s
]
if recent_fatigue:
avg_fatigue = sum(recent_fatigue) / len(recent_fatigue)
if avg_fatigue >= 7:
# Flag high-demand LOW priority events for postponement
for event in events:
if event.priority == EventPriority.LOW and event.requires_transport:
suggestions.append(SchedulingSuggestion(
suggestion_id=f"sug_{event.event_id}_fatigue",
original_event=event,
proposed_new_datetime=event.datetime_start + timedelta(days=7
reason=(
f"Your 7-day average fatigue score is {avg_fatigue:.1f}/1
f"'{event.title}' involves travel and physical demand. "
f"Rescheduling may improve treatment tolerance."
),
conflict_type=ConflictType.FATIGUE_WINDOW,
confidence=0.75,
scholarly_basis=(
"MOTOR (Steinberg et al., 2023): time-to-event risk strat
"for proactive patient management in chronic disease."
),
))
# Check lab values for ANC-based restrictions
if lab_values:
anc = lab_values.get("ANC") # Absolute Neutrophil Count (cells/µL)
if anc and anc < 1000:
for event in events:
if event.location and "hospital" not in event.location.lower():
suggestions.append(SchedulingSuggestion(
suggestion_id=f"sug_{event.event_id}_anc",
original_event=event,
proposed_new_datetime=None, # Defer — no proposed date yet
reason=(
f"Lab OCR detected ANC={anc} cells/µL (neutropenic threshold
f"Public venue events carry infection risk."
),
conflict_type=ConflictType.NEUTROPENIA_RISK,
confidence=0.91,
scholarly_basis=(
"Hantel et al. (2024, JAMA Netw Open): ethical AI in cancer s
"must incorporate lab-based clinical guardrails."
),
))
return suggestions
def build_transparency_report(
self, suggestions: List[SchedulingSuggestion]
) -> str:
"""
Generate a Glass Box audit trail for every AI scheduling decision.
Required by the 'model facts labels' standard (Carey, 2024, Advances in Consumer Rese
"""
lines = ["=== CALENDAR AI DECISION TRANSPARENCY REPORT ===\n"]
for s in suggestions:
lines.append(f"Suggestion ID : {s.suggestion_id}")
lines.append(f"Event : {s.original_event.title}")
lines.append(f"Conflict Type : {s.conflict_type.value}")
lines.append(f"Confidence : {s.confidence:.0%}")
lines.append(f"Reason : {s.reason}")
lines.append(f"Clinical Basis: {s.scholarly_basis}")
lines.append(f"Status : {s.status}")
lines.append("-" * 50)
return "\n".join(lines)

def patient_approval_node(state: AdvocateCalendarState) -> dict:
"""
HITL Gate — Phase 2/3.
In production: sends push notification to React Native app.
Patient taps Approve/Reject; response routes back here via webhook.
For simulation: auto-approves LOW priority, holds HIGH for human.
RLHF hook: rejected suggestions are stored in rejected_changes
to retrain the Master Model's scheduling preferences over time.
"""
print("--- Patient Approval Gate ---")
pending = state.get("pending_suggestions", [])
approved = []
rejected = []
held_for_human = []
for s in pending:
if s["confidence"] >= 0.85 and s["conflict_type"] in [
ConflictType.NEUTROPENIA_RISK.value,
ConflictType.DRUG_INTERACTION.value,
]:
# High-confidence clinical safety suggestions: flag for IMMEDIATE human review
held_for_human.append(s)
print(f" HELD FOR HUMAN: {s['event_title']} — {s['conflict_type']}")
else:
# Simulate patient approval (in prod: push notification response)
approved.append({**s, "status": "approved", "approved_at": datetime.now().isoform
print(f" Simulated approval: {s['event_title']}")
audit_entry = (
f"Approval Gate: {len(approved)} approved, "
f"{len(held_for_human)} held for human review, "
f"{len(rejected)} rejected."
)
return {
"approved_changes": approved,
"rejected_changes": rejected,
"pending_suggestions": held_for_human, # Remaining = still needs human
"audit_log": [audit_entry],
"next_agent": "ethics_screener" if held_for_human else "end",
}
def ethics_screener_node(state: AdvocateCalendarState) -> dict:
"""
Phase 3/4: Screens scheduling patterns for systemic ethics violations.
If the AI detects patterns suggesting insurance-driven care denial,
discriminatory scheduling, or repeated medically unjustified delays,
it flags items for the legal action queue (Phase 4).
Grounded in:
- EU AI Act high-risk classification for health AI (2024)
- Carey (2024): audit logs as basis for ex post complaint/litigation
- WHO (2024): multi-stakeholder governance for LMM health models
"""
print("--- Ethics & Legal Screener Activated ---")
held = state.get("pending_suggestions", [])
ethics_flags = list(state.get("ethics_flags", []))
legal_queue = list(state.get("legal_action_queue", []))
new_flags = []
for s in held:
# Flag: repeated high-confidence safety suggestions being blocked
if s["confidence"] > 0.85 and s["status"] == "pending":
flag = (
f"ETHICS FLAG [{datetime.now().isoformat()}]: "
f"High-confidence clinical safety suggestion blocked for '{s['event_title']}'
f"Conflict: {s['conflict_type']}. If this reflects insurance or provider obst
f"this event is queued for potential ethics complaint filing."
)
new_flags.append(flag)
# Pre-populate legal action queue (Phase 4 — filing module to be built)
legal_queue.append({
"action_type": "ethics_complaint",
"subject": s["event_title"],
"basis": s["reason"],
"scholarly_support": s["scholarly_basis"],
"timestamp": datetime.now().isoformat(),
"status": "queued",
"next_step": "human_attorney_review", # Always requires attorney sign-off
})
audit_entry = (
f"Ethics Screener: {len(new_flags)} new ethics flags raised. "
f"Legal action queue now has {len(legal_queue)} items."
)
msg = (
" Ethics Review Complete.\n"
+ ("\n".join(new_flags) if new_flags else "No ethics violations detected.")
+ (
f"\n\n{len(legal_queue)} item(s) queued for potential legal action. "
"An attorney must review before any complaint or motion is filed."
if legal_queue else ""
)
)
return {
"messages": [AIMessage(content=msg)],
"ethics_flags": ethics_flags + new_flags,
"legal_action_queue": legal_queue,
"audit_log": [audit_entry],
"next_agent": "end",
}
# ─────────────────────────────────────────────
# 5. BUILD THE GRAPH
# ─────────────────────────────────────────────
def calendar_router(state: AdvocateCalendarState) -> str:
return state.get("next_agent", "end")
def build_calendar_graph() -> StateGraph:
workflow = StateGraph(AdvocateCalendarState)
workflow.add_node("calendar_agent", calendar_agent_node)
workflow.add_node("patient_approval", patient_approval_node)
workflow.add_node("ethics_screener", ethics_screener_node)
workflow.set_entry_point("calendar_agent")
workflow.add_edge("calendar_agent", "patient_approval")
workflow.add_conditional_edges(
"patient_approval",
calendar_router,
{
"ethics_screener": "ethics_screener",
"end": END,
},
)
workflow.add_edge("ethics_screener", END)
return workflow.compile()
